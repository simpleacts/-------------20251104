/**
 * Validates and sanitizes table name to prevent SQL injection
 */
function validateTableName($tableName) {
    if (!preg_match('/^[a-zA-Z0-9_-]+$/', $tableName)) {
        throw new InvalidArgumentException("Invalid table name: '{$tableName}' contains invalid characters");
    }
    return $tableName;
}

/**
 * Fetches table data directly from database
 */
function fetchDirectTable($pdo, $tableName) {
    try {
        $safeTableName = validateTableName($tableName);
        
        // Check if table exists using schema_utils function (more reliable)
        if (!tableExists($pdo, $safeTableName)) {
            error_log("[data-io-data.php] Table '{$safeTableName}' does not exist in database");
            return [];
        }
        
        // Full data fetch - table name is validated, so safe to use
        $stmt = $pdo->query("SELECT * FROM `{$safeTableName}`");
        $rows = $stmt->fetchAll(PDO::FETCH_ASSOC);
        $rawCount = count($rows);
        error_log("[data-io-data.php] Fetched {$rawCount} raw rows from table '{$safeTableName}'");
        
        // デバッグ: 最初の数行のサンプルをログに出力
        if ($rawCount > 0) {
            $sampleRow = $rows[0];
            $sampleKeys = array_keys($sampleRow);
            $sampleValues = array_slice($sampleRow, 0, 5); // 最初の5カラムのみ
            error_log("[data-io-data.php] Sample row from '{$safeTableName}': " . json_encode($sampleValues, JSON_UNESCAPED_UNICODE));
        }
        
        // Filter out completely empty rows (but be more lenient)
        // すべての値がnullまたは空文字列の場合のみ除外
        $filteredRows = array_filter($rows, function($row) {
            if (!$row || !is_array($row)) {
                return false;
            }
            
            // 少なくとも1つのカラムに有効な値があるかチェック
            $hasValidValue = false;
            foreach ($row as $key => $value) {
                // null、空文字列、空配列以外は有効とみなす
                if ($value === null || $value === '') {
                    continue;
                }
                // 文字列の場合、空白のみでない場合は有効
                if (is_string($value) && trim($value) !== '') {
                    $hasValidValue = true;
                    break;
                }
                // 数値、真偽値、配列（空でない）は有効
                if (is_bool($value) || is_numeric($value) || (is_array($value) && count($value) > 0)) {
                    $hasValidValue = true;
                    break;
                }
            }
            
            return $hasValidValue;
        });
        
        $filteredCount = count($filteredRows);
        if ($filteredCount !== $rawCount) {
            error_log("[data-io-data.php] Filtered " . ($rawCount - $filteredCount) . " empty rows from table '{$safeTableName}' (kept {$filteredCount} rows)");
        } else {
            error_log("[data-io-data.php] No rows filtered from table '{$safeTableName}' (all {$rawCount} rows kept)");
        }
        
        return array_values($filteredRows);
    } catch (InvalidArgumentException $e) {
        error_log("[data-io-data.php] Invalid table name error: " . $e->getMessage());
        throw $e;
    } catch (PDOException $e) {
        error_log("[data-io-data.php] Error fetching table '{$tableName}': " . $e->getMessage());
        throw $e;
    }
}
